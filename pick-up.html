D3 Select

                
Everything in D3 is built off of the select function, which operates similarly to document.querySelector()

                
From there, you can manipulate properties.

                
                    
<html>
                    
<body>
                    
   <svg height=500 width=500>
                    
       <circle r="10" fill="red" id="red"></circle>
                    
       <circle r="20" fill="yellow" id="yellow"></circle>
                    
       <circle r="30" fill="green" id="green"></circle>
                    
       <circle r="40" fill="blue" id="blue"></circle>
                    
   </svg>
                    
</body>
                    
<script src="http://d3js.org/d3.v4.0.0-alpha.22.min.js"></script>
                    
<script>
                    
   var colors = ['red', 'yellow', 'green', 'blue']
                    
   colors.forEach(function(color, index) {
                    
       d3.select(`#${color}`)
                    
         .attr('cx', index * 100 + 50)
                    
         .attr('cy', index * 100 + 50)
                    
   })
                    
</script>
                    
</html>
