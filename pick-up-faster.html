D3 Select - Select all, append, data

                
You can grab entire groups of items and manipulate them with d3.

                
You can even create them on the fly, feeding them data to manage their properties.

                
                    
<html>
                    
<body>
                    
   <svg height=500 width=500>
                    
   </svg>
                    
</body>
                    
<script src="http://d3js.org/d3.v4.0.0-alpha.22.min.js"></script>
                    
<script>
                    
   const data = [{x: 20, y:20, r: 5, color: "red"}, {x: 45, y:75, r: 10, color: "yellow"}, {x: 70, y:20, r: 20, color: "green"}, {x: 95, y:120, r: 15, color: "blue"}, {x: 20, y:180, r: 20, color: "purple"}];
                    
   const g = d3.select("svg")
                    
       .append('g')
                    
       .selectAll('circle')
                    
       .data(data) 
                    
       .enter()
                    
       .append("circle") //appends a circle for each item in the data array
                    
       .attr("cx", function(d){ return d.x })
                    
       .attr("cy", function(d){ return d.y })
                    
       .attr("r", function(d){ return d.r })
                    
       .attr("fill", function(d){ return d.color });
                    
   d3.select('g')
                    
       .attr('transform', 'translate(200, 200)')
                    
</script>
                    
</html>
